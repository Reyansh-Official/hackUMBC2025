API Endpoints

POST /api/user/signup

// Request
{
  "email": "user@example.com",
  "password": "securepassword"
}

// Response
{
  "success": true,
  "session_id": "session_12345",
  "user": {
    "id": "user_12345",
    "email": "user@example.com",
    "created_at": "2023-05-15T10:30:00Z",
    "last_login": "2023-05-15T10:30:00Z",
    "interests": [],
    "modules": [],
    "badges": [],
    "progress_percentage": 0,
    "focus_sessions": [],
    "settings": {}
  }
}

POST /api/user/interests

// Request
{
  "user_id": "user_12345",
  "interests": ["Investing", "Budgeting", "Retirement Planning"]
}

// Response
{
  "success": true,
  "message": "Interests updated successfully",
  "user": {
    "id": "user_12345",
    "interests": ["Investing", "Budgeting", "Retirement Planning"],
    // Other user data...
  }
}

POST /api/user/upload-syllabus

// Request (multipart/form-data)
{
  "userId": "user_12345",
  "syllabus": [File]
}

// Response
{
  "success": true,
  "message": "Syllabus processed successfully",
  "modules_generated": 5
}
